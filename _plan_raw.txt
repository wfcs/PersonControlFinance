## Architecture Summary (Current State)

The codebase follows a clean layered architecture:

```
backend/
  app/
    api/v1/        # FastAPI routers (thin controllers)
    core/          # config.py, deps.py, security.py
    db/            # session.py (async engine + Base)
    models/        # SQLAlchemy 2.0 mapped models
    schemas/       # Pydantic v2 request/response DTOs
    services/      # Business logic (plain async functions)
    workers/       # Celery app + tasks
  migrations/      # Alembic
  tests/           # pytest-asyncio + httpx AsyncClient
```

**Key Conventions to Follow:**
- Services are plain `async def` functions, not classes (except the Pluggy client, which needs state for auth tokens)
- Every query filters by `tenant_id` for multi-tenant isolation
- Schemas use `model_config = {"from_attributes": True}` for ORM serialization
- Tests use SQLite in-memory with `dependency_overrides`
- `AuthUser` and `DBSession` are `Annotated` type aliases from `deps.py`
- Soft delete for accounts (`is_active = False`), hard delete for transactions
- `external_id` on Transaction is `unique=True` and indexed -- critical for Pluggy dedup

```python
import httpx
from datetime import datetime, timezone, timedelta
from app.core.config import settings

PLUGGY_BASE_URL = "https://api.pluggy.ai"

class PluggyClient:
    """Async wrapper for the Pluggy REST API."""

    def __init__(self) -> None:
        self._api_key: str | None = None
        self._api_key_expires_at: datetime | None = None
        self._http: httpx.AsyncClient = httpx.AsyncClient(
            base_url=PLUGGY_BASE_URL,
            timeout=30.0,
        )

    async def _ensure_api_key(self) -> str:
        """Authenticate with POST /auth and cache the key for ~1h50m."""
        now = datetime.now(timezone.utc)
        if self._api_key and self._api_key_expires_at and now < self._api_key_expires_at:
            return self._api_key
        resp = await self._http.post("/auth", json={
            "clientId": settings.PLUGGY_CLIENT_ID,
            "clientSecret": settings.PLUGGY_CLIENT_SECRET,
        })
        resp.raise_for_status()
        self._api_key = resp.json()["apiKey"]
        # Pluggy keys expire after 2h; renew at 1h50m to be safe
        self._api_key_expires_at = now + timedelta(hours=1, minutes=50)
        return self._api_key

    @property
    async def _headers(self) -> dict[str, str]:
        key = await self._ensure_api_key()
        return {"X-API-KEY": key, "Content-Type": "application/json"}

    async def create_connect_token(self, item_id: str | None = None,
                                     webhook_url: str | None = None,
                                     client_user_id: str | None = None) -> dict:
        """POST /connect_token -- used by frontend to open Connect Widget."""
        headers = await self._headers
        body: dict = {}
        if item_id:
            body["itemId"] = item_id
        if webhook_url:
            body["webhookUrl"] = webhook_url
        if client_user_id:
            body["clientUserId"] = client_user_id
        body["options"] = {"webhookUrl": webhook_url} if webhook_url else {}
        resp = await self._http.post("/connect_token", json=body, headers=headers)
        resp.raise_for_status()
        return resp.json()  # {"accessToken": "..."}

    async def get_item(self, item_id: str) -> dict:
        """GET /items/{item_id}"""
        headers = await self._headers
        resp = await self._http.get(f"/items/{item_id}", headers=headers)
        resp.raise_for_status()
        return resp.json()

    async def delete_item(self, item_id: str) -> None:
        """DELETE /items/{item_id}"""
        headers = await self._headers
        resp = await self._http.delete(f"/items/{item_id}", headers=headers)
        resp.raise_for_status()

    async def get_accounts(self, item_id: str) -> list[dict]:
        """GET /accounts?itemId={item_id}"""
        headers = await self._headers
        resp = await self._http.get("/accounts", params={"itemId": item_id}, headers=headers)
        resp.raise_for_status()
        return resp.json()["results"]

    async def get_transactions(self, account_id: str, date_from: str | None = None,
                                date_to: str | None = None, page: int = 1,
                                page_size: int = 500) -> dict:
        """GET /transactions?accountId={account_id}&from=...&to=..."""
        headers = await self._headers
        params: dict = {"accountId": account_id, "page": page, "pageSize": page_size}
        if date_from:
            params["from"] = date_from
        if date_to:
            params["to"] = date_to
        resp = await self._http.get("/transactions", params=params, headers=headers)
        resp.raise_for_status()
        return resp.json()  # {"total", "totalPages", "page", "results": [...]}

    async def get_transactions_by_ids(self, ids: list[str]) -> dict:
        """GET /transactions?ids={id1,id2,...}"""
        headers = await self._headers
        resp = await self._http.get("/transactions", params={"ids": ",".join(ids)}, headers=headers)
        resp.raise_for_status()
        return resp.json()

    async def close(self) -> None:
        await self._http.aclose()


# Singleton for app-wide reuse
pluggy_client = PluggyClient()
```

**Key Design Decisions:**
- Uses `httpx.AsyncClient` as required by the constraints
- Caches API key with a safety margin (1h50m vs 2h expiry)
- Singleton pattern via module-level instance
- All methods return raw `dict` -- Pydantic parsing happens in the service layer
- `close()` method for graceful shutdown (hook into FastAPI lifespan)

```python
from uuid import UUID
from datetime import datetime, timezone
from decimal import Decimal

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.integrations.pluggy_client import pluggy_client
from app.models.account import Account
from app.models.transaction import Transaction
from app.core.config import settings

PLUGGY_WEBHOOK_URL = f"{settings.SUPABASE_URL}/functions/v1/pluggy-webhook"
# Or better: a dedicated config setting

async def create_connect_token(
    tenant_id: UUID, user_id: UUID, db: AsyncSession,
    item_id: str | None = None,
) -> dict:
    """Generate a Pluggy Connect Token for the frontend widget."""
    webhook_url = settings.PLUGGY_WEBHOOK_URL  # New setting needed
    client_user_id = f"{tenant_id}:{user_id}"
    result = await pluggy_client.create_connect_token(
        item_id=item_id,
        webhook_url=webhook_url,
        client_user_id=client_user_id,
    )
    return result

async def sync_item_accounts(
    item_id: str, tenant_id: UUID, db: AsyncSession,
) -> list[Account]:
    """Fetch accounts from Pluggy and upsert into local DB."""
    pluggy_accounts = await pluggy_client.get_accounts(item_id)
    synced = []
    for pa in pluggy_accounts:
        # Find existing account by pluggy_account_id
        acc = await db.scalar(
            select(Account).where(
                Account.pluggy_account_id == pa["id"],
                Account.tenant_id == tenant_id,
            )
        )
        if acc:
            # Update balance and metadata
            acc.balance = Decimal(str(pa.get("balance", 0)))
            acc.name = pa.get("name", acc.name)
            acc.credit_limit = Decimal(str(pa["creditLimit"])) if pa.get("creditLimit") else None
        else:
            # Create new account
            acc = Account(
                tenant_id=tenant_id,
                name=pa.get("name", "Conta"),
                institution_name=pa.get("connector", {}).get("name", "Instituição"),
                institution_logo_url=pa.get("connector", {}).get("imageUrl"),
                type=_map_account_type(pa.get("type", "BANK")),
                pluggy_item_id=item_id,
                pluggy_account_id=pa["id"],
                balance=Decimal(str(pa.get("balance", 0))),
                credit_limit=Decimal(str(pa["creditLimit"])) if pa.get("creditLimit") else None,
            )
            db.add(acc)
        await db.flush()
        synced.append(acc)
    return synced

async def sync_item_transactions(
    item_id: str, tenant_id: UUID, db: AsyncSession,
    date_from: str | None = None,
) -> int:
    """Fetch transactions from Pluggy for all accounts of an item and upsert."""
    accounts = await db.scalars(
        select(Account).where(
            Account.pluggy_item_id == item_id,
            Account.tenant_id == tenant_id,
            Account.is_active.is_(True),
        )
    )
    total_synced = 0
    for acc in accounts.all():
        if not acc.pluggy_account_id:
            continue
        total_synced += await _sync_account_transactions(
            acc, tenant_id, db, date_from=date_from
        )
    return total_synced

async def _sync_account_transactions(
    account: Account, tenant_id: UUID, db: AsyncSession,
    date_from: str | None = None,
) -> int:
    """Paginate through all transactions for a single account."""
    page = 1
    synced = 0
    while True:
        result = await pluggy_client.get_transactions(
            account_id=account.pluggy_account_id,
            date_from=date_from,
            page=page,
        )
        for pt in result.get("results", []):
            ext_id = pt["id"]
            # Dedup by external_id (unique constraint)
            existing = await db.scalar(
                select(Transaction).where(Transaction.external_id == ext_id)
            )
            if existing:
                # Update if amount or description changed
                existing.amount = Decimal(str(pt["amount"]))
                existing.description = pt.get("description", existing.description)
            else:
                tx = Transaction(
                    tenant_id=tenant_id,
                    account_id=account.id,
                    description=pt.get("description", ""),
                    amount=Decimal(str(pt["amount"])),
                    type=_map_tx_type(pt.get("type", "DEBIT")),
                    date=datetime.fromisoformat(pt["date"]),
                    external_id=ext_id,
                    installment_number=_extract_installment_number(pt),
                    installment_total=_extract_installment_total(pt),
                )
                db.add(tx)
                synced += 1
        await db.flush()
        if page >= result.get("totalPages", 1):
            break
        page += 1
    return synced

def _map_account_type(pluggy_type: str) -> str:
    mapping = {
        "BANK": "checking",
        "CREDIT": "credit_card",
        "SAVINGS": "savings",
        "INVESTMENT": "investment",
    }
    return mapping.get(pluggy_type, "checking")

def _map_tx_type(pluggy_type: str) -> str:
    mapping = {"DEBIT": "debit", "CREDIT": "credit"}
    return mapping.get(pluggy_type, "debit")

def _extract_installment_number(pt: dict) -> int | None:
    meta = pt.get("creditCardMetadata")
    if meta and meta.get("installmentNumber"):
        return meta["installmentNumber"]
    return None

def _extract_installment_total(pt: dict) -> int | None:
    meta = pt.get("creditCardMetadata")
    if meta and meta.get("totalInstallments"):
        return meta["totalInstallments"]
    return None
```

```python
async def disconnect_item(
    item_id: str, current_user: AuthUser, db: DBSession
) -> None:
    """Disconnect a Pluggy item and soft-delete its accounts."""
    # Verify ownership
    from sqlalchemy import select
    from app.models.account import Account
    accounts = await db.scalars(
        select(Account).where(
            Account.pluggy_item_id == item_id,
            Account.tenant_id == current_user.tenant_id,
        )
    )
    acc_list = accounts.all()
    if not acc_list:
        raise HTTPException(status_code=404, detail="Item not found")
    # Soft delete all accounts for this item
    for acc in acc_list:
        acc.is_active = False
    # Delete from Pluggy
```

```python
import logging
from fastapi import APIRouter, BackgroundTasks, Request, Response, status
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from app.db.session import AsyncSessionLocal
from app.models.account import Account
from app.services import pluggy_service
from app.schemas.pluggy import PluggyWebhookPayload

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/webhooks", tags=["Webhooks"])

@router.post("/pluggy", status_code=status.HTTP_200_OK)
async def pluggy_webhook(request: Request, background_tasks: BackgroundTasks) -> Response:
    """
    Receive Pluggy webhook notifications.
    MUST return 2XX immediately per Pluggy docs, then process async.
    """
    body = await request.json()
    # Convert camelCase to snake_case (Pluggy sends camelCase)
    payload = _parse_pluggy_payload(body)
    logger.info(f"Pluggy webhook: event={payload.event}, item_id={payload.item_id}")

    # Process in background to return 200 immediately
    if payload.event in ("item/created", "item/updated"):
        background_tasks.add_task(_handle_item_event, payload)
    elif payload.event == "transactions/created":
        background_tasks.add_task(_handle_transactions_created, payload)
    elif payload.event == "transactions/updated":
        background_tasks.add_task(_handle_transactions_updated, payload)
    elif payload.event in ("item/error", "item/login_error"):
        background_tasks.add_task(_handle_item_error, payload)

    return Response(status_code=200)

def _parse_pluggy_payload(body: dict) -> PluggyWebhookPayload:
    """Map camelCase Pluggy payload to snake_case Pydantic model."""
    return PluggyWebhookPayload(
        event=body.get("event", ""),
        event_id=body.get("eventId"),
        item_id=body.get("itemId"),
        account_id=body.get("accountId"),
        triggered_by=body.get("triggeredBy"),
        client_user_id=body.get("clientUserId"),
        transactions_count=body.get("transactionsCount"),
        created_transactions_link=body.get("createdTransactionsLink"),
        transaction_ids=body.get("transactionIds"),
    )

async def _handle_item_event(payload: PluggyWebhookPayload) -> None:
    """When item is created/updated, sync accounts + transactions."""
    if not payload.item_id or not payload.client_user_id:
        return
    # client_user_id format: "{tenant_id}:{user_id}"
    tenant_id_str = payload.client_user_id.split(":")[0]
    from uuid import UUID
    tenant_id = UUID(tenant_id_str)

    async with AsyncSessionLocal() as db:
        try:
            await pluggy_service.sync_item_accounts(payload.item_id, tenant_id, db)
            await pluggy_service.sync_item_transactions(payload.item_id, tenant_id, db)
            await db.commit()
        except Exception:
            await db.rollback()
            logger.exception(f"Failed to sync item {payload.item_id}")

async def _handle_transactions_created(payload: PluggyWebhookPayload) -> None:
    """New transactions created -- use createdTransactionsLink or sync account."""
    if not payload.item_id or not payload.client_user_id:
        return
    tenant_id = UUID(payload.client_user_id.split(":")[0])
    async with AsyncSessionLocal() as db:
        try:
            await pluggy_service.sync_item_transactions(
                payload.item_id, tenant_id, db,
                date_from=None,  # Full sync or use transactionsMinDate
            )
            await db.commit()
        except Exception:
            await db.rollback()
            logger.exception(f"Failed to sync transactions for item {payload.item_id}")

async def _handle_transactions_updated(payload: PluggyWebhookPayload) -> None:
    """Transactions updated -- fetch by IDs and update."""
    if not payload.transaction_ids or not payload.client_user_id:
        return
    from app.integrations.pluggy_client import pluggy_client
    from app.models.transaction import Transaction
    from decimal import Decimal

    tenant_id = UUID(payload.client_user_id.split(":")[0])
    result = await pluggy_client.get_transactions_by_ids(payload.transaction_ids)
    async with AsyncSessionLocal() as db:
        try:
            for pt in result.get("results", []):
                tx = await db.scalar(
                    select(Transaction).where(
                        Transaction.external_id == pt["id"],
                        Transaction.tenant_id == tenant_id,
                    )
                )
                if tx:
                    tx.amount = Decimal(str(pt["amount"]))
                    tx.description = pt.get("description", tx.description)
            await db.commit()
        except Exception:
            await db.rollback()
            logger.exception("Failed to update transactions")

async def _handle_item_error(payload: PluggyWebhookPayload) -> None:
    """Log item errors for monitoring."""
    logger.error(f"Pluggy item error: item_id={payload.item_id}, event={payload.event}")
```

#### 3.1 -- Implement Task Stubs (`backend/app/workers/tasks.py`)

Replace the current stubs with working implementations. Celery tasks are synchronous, so we need `asyncio.run()` to call async services.

```python
import asyncio
import logging
from uuid import UUID
from app.workers.celery_app import celery_app
from app.db.session import AsyncSessionLocal

logger = logging.getLogger(__name__)

@celery_app.task(name="app.workers.tasks.sync_all_tenants", bind=True, max_retries=3)
def sync_all_tenants(self) -> None:
    """Dispatch sync for all tenants with active Pluggy connections."""
    asyncio.run(_sync_all_tenants_async())

async def _sync_all_tenants_async() -> None:
```

### BE-07: Recurrence Detection Engine

**Depends on:** BE-04 (transactions must exist in DB)

#### 4.1 -- Recurrence Service (`backend/app/services/recurrence_service.py`)

**New file: `backend/app/services/recurrence_service.py`**

The detection algorithm:
1. **Installments:** Look for transactions with `installment_number` and `installment_total` already set by Pluggy (via `creditCardMetadata`). Group them by description pattern similarity.
2. **Fixed Recurring:** Analyze 3+ months of history. Group transactions by similar description (fuzzy match) and similar amounts. If they appear monthly with consistent amounts, flag as recurring.

```python
from uuid import UUID
from datetime import date, timedelta
from decimal import Decimal
from collections import defaultdict

from sqlalchemy import select, func, and_
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.transaction import Transaction
from app.models.recurrence import Recurrence
from app.schemas.recurrence import RecurrenceCreate, RecurrenceOut

async def detect_recurrences(tenant_id: UUID, db: AsyncSession) -> list[Recurrence]:
    """
    Analyze transaction history to detect installments and recurring expenses.
    Returns list of newly created/updated Recurrence records.
    """
    recurrences = []
    recurrences.extend(await _detect_installments(tenant_id, db))
    recurrences.extend(await _detect_fixed_recurring(tenant_id, db))
    await db.flush()
    return recurrences

async def _detect_installments(tenant_id: UUID, db: AsyncSession) -> list[Recurrence]:
    """Detect installment series from transactions with installment_number set."""
    result = await db.execute(
        select(Transaction)
        .where(
            Transaction.tenant_id == tenant_id,
            Transaction.installment_total.isnot(None),
            Transaction.recurrence_id.is_(None),  # Not yet linked
        )
        .order_by(Transaction.description, Transaction.installment_number)
    )
    transactions = result.scalars().all()

    # Group by normalized description + total installments + similar amount
    groups = defaultdict(list)
    for tx in transactions:
        key = (_normalize_desc(tx.description), tx.installment_total, _round_amount(tx.amount))
        groups[key].append(tx)

    recurrences = []
    for (desc, total, amount), txs in groups.items():
        if len(txs) < 2:
            continue
        max_installment = max(tx.installment_number or 0 for tx in txs)
        last_date = max(tx.date for tx in txs)
        next_due = last_date.date() + timedelta(days=30) if max_installment < total else None

        rec = Recurrence(
            tenant_id=txs[0].tenant_id,
            description=desc,
            amount=abs(amount),
            frequency="installment",
            total_installments=total,
            current_installment=max_installment,
            next_due_date=next_due if next_due and max_installment < total else None,
            is_active=max_installment < total,
        )
        db.add(rec)
        await db.flush()

        for tx in txs:
            tx.recurrence_id = rec.id
        recurrences.append(rec)

    return recurrences

async def _detect_fixed_recurring(tenant_id: UUID, db: AsyncSession) -> list[Recurrence]:
    """
    Detect fixed recurring expenses by finding transactions that repeat
    monthly with similar descriptions and amounts.
    Requires 3+ occurrences to classify as recurring.
    """
    # Get all expense transactions from last 6 months
    six_months_ago = date.today() - timedelta(days=180)
    result = await db.execute(
        select(Transaction)
        .where(
            Transaction.tenant_id == tenant_id,
            Transaction.amount < 0,  # expenses only
            Transaction.date >= six_months_ago,
            Transaction.recurrence_id.is_(None),
        )
        .order_by(Transaction.description, Transaction.date)
    )
    transactions = result.scalars().all()

    # Group by normalized description
    groups = defaultdict(list)
    for tx in transactions:
        key = _normalize_desc(tx.description)
        groups[key].append(tx)

    recurrences = []
    for desc, txs in groups.items():
        if len(txs) < 3:
            continue
        # Check if amounts are similar (within 10% variance)
        amounts = [abs(tx.amount) for tx in txs]
        avg_amount = sum(amounts) / len(amounts)
        if any(abs(a - avg_amount) / avg_amount > 0.1 for a in amounts):
            continue
        # Check if roughly monthly (25-35 day intervals)
        dates = sorted(tx.date for tx in txs)
        intervals = [(dates[i+1] - dates[i]).days for i in range(len(dates)-1)]
        avg_interval = sum(intervals) / len(intervals)
        if not (20 <= avg_interval <= 40):
            continue

        last_date = max(tx.date for tx in txs)
        next_due = last_date.date() + timedelta(days=int(avg_interval))

        rec = Recurrence(
            tenant_id=txs[0].tenant_id,
            description=desc,
            amount=avg_amount,
            frequency="monthly",
            next_due_date=next_due,
            is_active=True,
        )
        db.add(rec)
        await db.flush()
        for tx in txs:
            tx.recurrence_id = rec.id
        recurrences.append(rec)

    return recurrences

def _normalize_desc(desc: str) -> str:
    """Normalize description for grouping (remove dates, installment numbers)."""
    import re
    desc = desc.upper().strip()
    desc = re.sub(r'\d{1,2}/\d{1,2}', '', desc)  # Remove "1/12", "03/06"
    desc = re.sub(r'PARC\s*\d+', '', desc)  # Remove "PARC 3"
    desc = re.sub(r'\s+', ' ', desc).strip()
    return desc

def _round_amount(amount: Decimal) -> Decimal:
    """Round to nearest integer for grouping purposes."""
    return Decimal(str(round(abs(amount))))
```

#### 5.1 -- Bill Service (`backend/app/services/bill_service.py`)

**New file: `backend/app/services/bill_service.py`**

Credit card billing cycles: group transactions by card account and billing period. Most Brazilian credit cards close on a specific day of the month.

```python
from uuid import UUID
from datetime import datetime, date, timedelta
from decimal import Decimal
from sqlalchemy import select, func, and_
from sqlalchemy.ext.asyncio import AsyncSession
from app.models.transaction import Transaction
from app.models.account import Account

async def get_credit_card_bills(
    tenant_id: UUID, account_id: UUID, db: AsyncSession,
    closing_day: int = 25,  # Default closing day
) -> list[dict]:
    """
    Group credit card transactions by billing cycle.
    Returns list of bills with total, status, and transactions.
    """
    account = await db.scalar(
        select(Account).where(
            Account.id == account_id,
            Account.tenant_id == tenant_id,
            Account.type == "credit_card",
        )
    )
    if not account:
        from fastapi import HTTPException, status
        raise HTTPException(status_code=404, detail="Credit card account not found")

    # Get all transactions for this credit card
    transactions = (await db.scalars(
        select(Transaction).where(
            Transaction.account_id == account_id,
            Transaction.tenant_id == tenant_id,
            Transaction.is_excluded.is_(False),
        ).order_by(Transaction.date.desc())
    )).all()

    # Group by billing cycle
    bills: dict[str, dict] = {}
    for tx in transactions:
        bill_key = _get_bill_period(tx.date, closing_day)
        if bill_key not in bills:
            bills[bill_key] = {
                "period": bill_key,
                "closing_date": _get_closing_date(tx.date, closing_day).isoformat(),
                "due_date": (_get_closing_date(tx.date, closing_day) + timedelta(days=10)).isoformat(),
                "total": Decimal("0"),
                "transaction_count": 0,
                "transactions": [],
            }
        bills[bill_key]["total"] += tx.amount
        bills[bill_key]["transaction_count"] += 1

    return sorted(bills.values(), key=lambda b: b["period"], reverse=True)

def _get_bill_period(tx_date: datetime, closing_day: int) -> str:
    """Determine which billing period a transaction belongs to."""
    d = tx_date.date() if isinstance(tx_date, datetime) else tx_date
    if d.day <= closing_day:
        return f"{d.year}-{d.month:02d}"
    else:
        next_month = d.replace(day=1) + timedelta(days=32)
        return f"{next_month.year}-{next_month.month:02d}"

def _get_closing_date(tx_date: datetime, closing_day: int) -> date:
    d = tx_date.date() if isinstance(tx_date, datetime) else tx_date
    if d.day <= closing_day:
        return d.replace(day=closing_day)
    else:
        next_month = d.replace(day=1) + timedelta(days=32)
        return next_month.replace(day=closing_day)
```

```python
from uuid import UUID
from datetime import date, timedelta
from decimal import Decimal
from dateutil.relativedelta import relativedelta
from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession
from app.models.account import Account
from app.models.recurrence import Recurrence

async def project_balance(
    tenant_id: UUID, db: AsyncSession,
    months: int = 3,  # 3, 6, or 12
) -> list[dict]:
    """
    Project balance forward based on:
    - Current total balance across all accounts
    - Active recurrences (monthly expenses/income)
    - Installments remaining
    """
    # Get current total balance
    total_balance = await db.scalar(
        select(func.sum(Account.balance)).where(
            Account.tenant_id == tenant_id,
            Account.is_active.is_(True),
        )
    ) or Decimal("0")

    # Get active recurrences
    recurrences = (await db.scalars(
        select(Recurrence).where(
            Recurrence.tenant_id == tenant_id,
            Recurrence.is_active.is_(True),
        )
    )).all()

    projections = []
    current_balance = total_balance
    today = date.today()

    for i in range(1, months + 1):
        target_month = today + relativedelta(months=i)
        month_delta = Decimal("0")

        for rec in recurrences:
            if rec.frequency == "monthly":
                month_delta -= rec.amount  # expenses are stored as positive in recurrence
            elif rec.frequency == "installment":
                if rec.current_installment and rec.total_installments:
                    if rec.current_installment + i <= rec.total_installments:
                        month_delta -= rec.amount

        current_balance += month_delta
        projections.append({
            "month": target_month.strftime("%Y-%m"),
            "projected_balance": current_balance,
            "monthly_delta": month_delta,
        })

    return projections
```

```python
async def get_net_worth(tenant_id: UUID, db: AsyncSession) -> dict:
    """Assets vs liabilities, net worth."""

            Account.tenant_id == tenant_id,

        "liabilities": liabilities,

        "by_type": by_type,
```

```python
async def auto_categorize_transaction(
    tx: Transaction, tenant_id: UUID, db: AsyncSession,
) -> UUID | None:

            cat = await db.scalar(
```

## Key Risks and Mitigations

1. **Pluggy API rate limits:** Use the Celery beat schedule (30 min) to spread sync load. Add exponential backoff with `tenacity` (already in dependencies).

2. **Transaction dedup race conditions:** The `external_id` unique constraint is the safety net. Wrap upsert logic in a try/except for `IntegrityError` and skip duplicates.

3. **Celery + async mismatch:** Celery tasks are synchronous. Use `asyncio.run()` in each task to bridge the gap. This creates a new event loop per task execution, which is fine for background processing.

4. **Webhook security:** Pluggy webhooks have no HMAC signature like Stripe. Security relies on IP whitelisting (`177.71.238.212`) and optional custom headers. Add middleware to verify source IP in production.

5. **Multi-tenant data leaks:** Every new service function MUST include `tenant_id` in every query. The existing RLS policies provide a second layer of defense at the database level.
